echo "Add LIRC Support"
FILE_LIRCD="/etc/lirc/lircd.conf"
FILE_LIRCRC="/etc/lirc/lircrc"
FILE_HARDWARE="/etc/lirc/hardware.conf"
systemctl stop lirc
if [ -f "$FILE_LIRCD" ]; then
        rm $FILE_LIRCD
else
        touch $FILE_LIRCD
fi
if [ -f "$FILE_LIRCRC" ]; then
        rm $FILE_LIRCRC
else
        touch $FILE_LIRCRC
fi
        echo 'begin remote

  name  /etc/lirc/lircd.conf
  bits           16
  flags SPACE_ENC|CONST_LENGTH
  eps            30
  aeps          100

  header       9151  4540
  one           558  1700
  zero          558   611
  ptrail        555
  repeat       9124  2280
  pre_data_bits   16
  pre_data       0xFF
  gap          108820
  repeat 9000 2250
  toggle_bit_mask 0x0

      begin codes
          pause                    0xC23D
          nextsong                 0x02FD
          prevsong                 0x22DD
          mute                     0x906F
          volumeup                 0xA857
          volumedown               0xE01F
          stop                     0x629D
      end codes

end remote' > $FILE_LIRCD
echo 'begin
        prog = irexec
        button = pause
	delay = 2
	repeat = 3
        config =mpc toggle
end
begin
        prog = irexec
        button = nextsong
	delay = 2
        repeat = 3
        config = mpc next
end
begin
        prog = irexec
        button = prevsong
	delay = 2
        repeat = 3
        config = mpc prev
end
begin
        prog = irexec
        button = mute
	delay = 2
	repeat = 3
        config = mpc toggle
end
begin
        prog = irexec
        button = volumeup
	delay = 2
        repeat = 3
        config = mpc volume +5
end
begin
        prog = irexec
        button = volumedown
	delay = 2
	repeat = 3
        config = mpc volume -5
end' > $FILE_LIRCRC
sed -i "s/^LIRCD_ARGS=\".*\"/LIRCD_ARGS=\"--uinput\"/" /etc/lirc/hardware.conf
sed -i "s/^DRIVER=\".*\"/DRIVER=\"default\"/" /etc/lirc/hardware.conf
sed -i "s/^DEVICE=\".*\"/DEVICE=\"\/dev\/lirc0\"/" /etc/lirc/hardware.conf
sed -i "s/^MODULES=\".*\"/MODULES=\"lirc_rpi\"/" /etc/lirc/hardware.conf

echo "Enable lirc-rpi overlay in config.txt"
linenum=$(grep -n 'gpu_mem' build/scripts/raspberryconfig.sh | awk -F ":" '{print $1}')
linenumfinal=$[$linenum+1]
linenumfinala=$linenumfinal"a"
echo "Lirc in config.txt line is:"
echo $linenumfinala
sed -i "$linenumfinala dtoverlay=lirc-rpi,gpio_in_pin=26,gpio_in_pull=up" /boot/config.txt
